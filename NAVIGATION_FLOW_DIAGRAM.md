### Role-Based Initial Route Logic
```javascript
function getInitialRoute() {
  const userRoles = user?.roles || [];
  
  // Admin users вҶ’ AdminBoard
  if (userRoles.some(role => ['admin', 'admindoctor'].includes(role))) {
    return 'AdminBoard';
  }
  
  // All other users вҶ’ Appointments
  return 'Appointments';
}
```
